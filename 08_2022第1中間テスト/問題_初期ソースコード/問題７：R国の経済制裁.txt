####__弶婜僜乕僗僐乕僪__

#include <iostream>
using namespace std;

double open_rate = 100.0;
double darkness_rate = 20.0;

//拞宲崙
double relay(double doller, int num) {
	if (num == 0) {
		return(doller / darkness_rate);
	}
	else if (num == 1) {
		return(doller * 0.5 / open_rate + doller * 0.4 / darkness_rate);
	}
	else {
		return(0);
	}
}

int main() {
	double doller, gas;
	int num;		//拞宲崙悢
	cin >> doller >> num;
	gas = relay(doller, num);
	cout << gas;
}
